数据库设计

创建数据库

CREATE DATABASE yellowstar;

数据表设计根据需求，并且参照阿里巴巴Java开发手册 1.4.0中的相关规范 参照说明：

  • 建表规范9中，表必备三字段，id,gmt_create,gmt_modified

创建用户表

表名：yw_user

CREATE TABLE IF NOT EXISTS yw_user (
    uid INT UNSIGNED NOT NULL AUTO_INCREMENT KEY COMMENT 'UID',
    loginname VARCHAR(20) NOT NULL UNIQUE comment '登录名',
    username VARCHAR(20) NOT NULL UNIQUE comment '用户名',
    password CHAR(32) NOT NULL COMMENT '密码：md5加密',
    user_type TINYINT UNSIGNED NOT NULL COMMENT '用户类型',
    phone VARCHAR(11) NOT NULL comment '手机号码',
    is_valid TINYINT UNSIGNED NOT NULL COMMENT '是否可用',
    gmt_create DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL COMMENT '创建时间',
    gmt_modified DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL COMMENT '修改时间'
);

type说明：

  • 1:超级管理员
  • 2:物业
  • 3:业主

参照说明：

  • 建表规范9中，表必备三字段，id,gmt_create,gmt_modified

房产信息表

表名：yw_house

CREATE TABLE IF NOT EXISTS yw_house (
    hid INT UNSIGNED NOT NULL AUTO_INCREMENT KEY COMMENT '房产号',
    username VARCHAR(20) NOT NULL comment '房产持有人',
    house_type VARCHAR(20) NOT NULL comment '房产类型',
    area INT UNSIGNED NOT NULL COMMENT '住房面积',
    gmt_buy DATE NOT NULL COMMENT '购房时间',
    building VARCHAR(20) NOT NULL comment '楼号',
    unit VARCHAR(10) NOT NULL comment '小区单元',
    house_num VARCHAR(10) NOT NULL COMMENT '门牌号',
    gmt_create DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL COMMENT '创建时间',
    gmt_modified DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL COMMENT '修改时间'
);

停车位信息表

表名：yw_stall

CREATE TABLE IF NOT EXISTS yw_stall (
    id INT UNSIGNED NOT NULL AUTO_INCREMENT KEY COMMENT '停车号',
    location VARCHAR(20) NOT NULL comment '位置',
    licenseplatenumber VARCHAR(8) NOT NULL comment '车牌号',
    gmt_create DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL COMMENT '创建时间',
    gmt_modified DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL COMMENT '修改时间',
    hid INT UNSIGNED NOT NULL  COMMENT '外键房产id',
    FOREIGN KEY(hid) REFERENCES yw_house(hid)
);

费用单价表

表名：yw_perprice

CREATE TABLE IF NOT EXISTS yw_unitprice (
    pid INT UNSIGNED NOT NULL AUTO_INCREMENT KEY COMMENT '单价编号',
    cost_type TINYINT UNSIGNED NOT NULL comment '单价类型',
    unit_price DECIMAL(18,2) UNSIGNED NOT NULL comment '单价',
    gmt_create DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL COMMENT '创建时间',
    gmt_modified DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL COMMENT '修改时间'

p_type说明:

  • 1:水费
  • 2:电费
  • 3:燃气费

表名：yw_cost

CREATE TABLE IF NOT EXISTS yw_cost (
    id INT UNSIGNED NOT NULL AUTO_INCREMENT KEY COMMENT '编号',
    hid INT UNSIGNED NOT NULL  COMMENT '外键房产id',
    cost_type TINYINT UNSIGNED NOT NULL comment '单价类型',
    unit_price DECIMAL(18,2) UNSIGNED NOT NULL comment '单价',
    amount DECIMAL(18,2) UNSIGNED NOT NULL comment '用量',
    gmt_create DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL COMMENT '创建时间',
    gmt_modified DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL COMMENT '修改时间',
    FOREIGN KEY(hid) REFERENCES yw_house(hid)
);
